# 1. إضافة كل التغييرات (الملفات والفولدرات)
git add .

# 2. تسجيل التغييرات برسالة توضيحية
git commit -m "تحديث شامل للمشروع"

# 3. رفع التعديلات للسيرفر (مع فرض النسخة لو فيه تعارض)
git push -f origin main












# 1. إضافة الملف المحدد فقط (استبدل اسم الملف بالاسم اللي عندك)
git add index.html

# 2. تسجيل تعديل الملف
git commit -m "تعديل ملف الاندكس فقط"

# 3. الرفع للسيرفر
git push origin main
